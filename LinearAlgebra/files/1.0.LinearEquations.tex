\chapter{Vector Spaces and Linear Equations}
\section{Basic Structures and Definitions}
\defn{Ring $R$}{A set $R$ with two operations $+:R \times R \to R$ and $\cdot: R\times R\to R$ is called a ring if the following hold:
\begin{enumerate}
   \item (Additive abelian group) $(R,+)$ is an abelian group with identity $0$.
   \item (Distributivity) $x(a+b)=x\cdot a+x\cdot b$ and $(x+y)a=x\cdot a+y\cdot a$
\end{enumerate}
}

\defn{Field}{$(\FF,+,\cdot)$ is a field if $\FF$ is a set with operations $+:\FF \times\FF \to \FF$ and $\cdot: \FF\times \FF \to \FF$ such that the following hold:
\begin{enumerate}
\item $(\FF,+)$ is an abelian group with an identity element $0$, i.e. $\forall a,b,c \in \FF$, we have $a+(b+c)=(a+b)+c$ (Associativity), $a+b=b+a$ (Commutativity), $\exists 0 \in \FF$ such that $a+0=a$, and $\forall a \in \FF$, $\exists -a \in \FF$ such that $a+(-a)=0$.
\item $(\FF\setminus\{0\},\cdot)$ is also an abelian group, i.e. $\forall a,b,c \in \FF\setminus\{0\}$, we have $a\cdot(b\cdot c)=(a\cdot b)\cdot c$ (Associativity), $a\cdot b=b\cdot a$, $\exists 1$ such that $a\cdot 1=a$ and $\forall a \in \FF\setminus\{0\}$ there exists a unique $a^{-1}$ such that $a a^{-1}=1$.
\item (Distributivity) $\forall a,b,c \in \FF$, $a(b+c)=ab+ac$.
\end{enumerate}
Such a structure is called a \emph{Field}.
}
