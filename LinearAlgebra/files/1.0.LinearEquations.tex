\section{Fields}

\defn{Fields (Under construction)}{
   A field $\mathcal{F}$ is a set of objects that satisfy the following the properties.\\\\
   For any x,y $\in$ $\mathcal{F}$,
   \begin{enumerate}
      \item $x+y = y+x$ is true.
      \item $x+(x+y)=(x+y)+z$.
      \item $\exists$ a unique element.
   \end{enumerate} 
}
